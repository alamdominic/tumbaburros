Funciones def · lambda · args
estructura completa
def saludar(nombre: str,
            formal: bool = False) -> str:
    """Devuelve un saludo."""
    trato = "Sr/a" if formal else ""
    return f"Hola {trato} {nombre}"

# *args  y  **kwargs
def total(*nums, **opts):
    return sum(nums)
lambda · retorno múltiple
# función anónima
doble = lambda x: x * 2

# retorno múltiple
def min_max(lst):
    return min(lst), max(lst)

lo, hi = min_max([3,1,4,9])

Listas mutable · ordenada 
nums = [10, 20, 30, 40]
nums[0]         # 10 (índice)
nums[-1]        # 40 (último)
nums[1:3]       # [20,30] slice
nums.append(50)  # agrega al final
nums.remove(20)  # elimina valor
nums.pop(0)      # elimina índice
sorted(nums)    # nueva ordenada
nums.sort(reverse=True) # in-place
len(nums)       # longitud

Manejo de errores — try / except excepciones · robusto
estructura completa
try:
    val = int(input("Número: "))
    res = 10 / val
except ValueError:
    print("Ingresa un número")
except ZeroDivisionError:
    print("No entre ceros")
except Exception as e:
    print(f"Error: {e}")
else:
    print("Sin errores")
finally:
    print("Siempre ejecuta")
errores frecuentes + raise
ValueError         # tipo inválido
TypeError          # op. inválida
IndexError         # índice fuera
KeyError           # clave inexist.
FileNotFoundError  # archivo
AttributeError     # atributo
NameError          # var no definida

# lanzar excepción manual
raise ValueError("Dato inválido")
Nunca: except: pass — silencia errores y hace el debugging imposible.

Comprensiones one-liner elegante 
# lista
cuad = [x**2 for x in range(10)]

# con filtro
pares = [x for x in range(20)
         if x % 2 == 0]

# dict comprehension
d = {x: x**2 for x in range(5)}
# set comprehension
letras = {c.lower() for c in "Python"}

# generator (memoria eficiente)
gen = (x**2 for x in range(1000))
next(gen)   # 0  — lazy eval

# anidado (flatten)
plano = [n for fila in matriz
         for n in fila]
Legibilidad: Si supera 2 condiciones, usa un for normal.
